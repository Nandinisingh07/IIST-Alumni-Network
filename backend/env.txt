        const BACKEND_API = "http://127.0.0.1:8000";

        // Navigation Engine
        function navigateTo(id) {
            document.querySelectorAll('.app-section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('nav button').forEach(b => b.classList.remove('nav-active'));
            
            document.getElementById('sect-' + id).classList.add('active');
            document.getElementById('tab-' + id).classList.add('nav-active');

        // Helper: Pre-fill functions
        function sendChatFromBtn(txt) { document.getElementById('chatField').value = txt; sendChatApi(); }
        function setMentorSearch(txt) { document.getElementById('mentorInput').value = txt; fetchMentors(); }
        function genRoad(txt) { if(txt) document.getElementById('roadField').value = txt; runRoadmapApi(); }

        // --- MENTOR FETCH ---
        async function fetchMentors() {
            const skill = document.getElementById('mentorInput').value;
            const grid = document.getElementById('mentorGrid');
            grid.innerHTML = '<div class="col-span-full py-20 text-center"><i class="fas fa-spinner fa-spin text-5xl text-indigo-600"></i></div>';
            
            try {
                const res = await fetch(`${BACKEND_API}/mentor-match`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ skill })
                });
                const data = await res.json();
                grid.innerHTML = data.length ? '' : '<p class="col-span-full text-slate-400 font-bold text-center py-20">No mentors found for this skill yet.</p>';
                data.forEach(m => {
                    grid.innerHTML += `
                        <div class="bg-white p-8 rounded-[3rem] border border-slate-100 shadow-sm hover:shadow-2xl transition-all duration-500 group">
                            <div class="flex items-center gap-5 mb-8">
                                <div class="w-16 h-16 bg-indigo-950 text-white rounded-[1.5rem] flex items-center justify-center text-2xl font-black shadow-xl">${m.name[0]}</div>
                                <div><h4 class="text-xl font-black text-slate-900">${m.name}</h4><p class="text-indigo-600 font-bold text-xs uppercase">${m.role} @ ${m.company}</p></div>
                            </div>
                            <div class="flex flex-wrap gap-2 mb-8">${m.skills.split(',').map(s => `<span class="bg-slate-50 text-slate-500 text-[9px] font-black px-3 py-1.5 rounded-lg border border-slate-100 uppercase">${s.trim()}</span>`).join('')}</div>
                            <button class="w-full bg-slate-100 text-slate-900 py-4 rounded-2xl font-black text-xs hover:bg-indigo-950 hover:text-white transition shadow-sm active:scale-95">Connect Profile</button>
                        </div>
                    `;
                });
            } catch(e) { grid.innerHTML = '<div class="col-span-full text-red-500 font-bold text-center">Error: API unreachable</div>'; }
        }

        // --- CHAT API ---
        async function sendChatApi() {
            const input = document.getElementById('chatField');
            const win = document.getElementById('chatWindow');
            const msg = input.value.trim();
            if(!msg) return;

            win.innerHTML += `<div class="bg-white p-6 rounded-3xl rounded-tr-none max-w-[80%] self-end ml-auto shadow-sm border border-slate-100 font-semibold text-slate-700">${msg}</div>`;
            input.value = '';
            
            const loadingId = 'L' + Date.now();
            win.innerHTML += `<div id="${loadingId}" class="bg-indigo-950 text-indigo-200/50 p-6 rounded-3xl rounded-tl-none max-w-[80%] shadow-xl italic font-bold">AI Mentor is analyzing...</div>`;
            win.scrollTop = win.scrollHeight;

            try {
                const res = await fetch(`${BACKEND_API}/chat`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ message: msg })
                });
                const data = await res.json();
                document.getElementById(loadingId).remove();
                win.innerHTML += `<div class="bg-indigo-950 text-white p-6 rounded-3xl rounded-tl-none max-w-[85%] shadow-2xl leading-relaxed">${data.response}</div>`;
            } catch(e) { document.getElementById(loadingId).innerHTML = '<span class="text-red-400">Connection Failed: Start Backend</span>'; }
            win.scrollTop = win.scrollHeight;
        }

        // --- ROADMAP API ---
        async function runRoadmapApi() {
            const role = document.getElementById('roadField').value;
            const resBox = document.getElementById('roadDisplay');
            if(!role) return;
            resBox.innerHTML = '<div class="py-10 text-center"><i class="fas fa-circle-notch fa-spin text-5xl text-indigo-600"></i></div>';

            try {
                const res = await fetch(`${BACKEND_API}/career-roadmap`, {
                    method: 'POST',
                    headers: {'Content-Type': 'application/json'},
                    body: JSON.stringify({ role })
                });
                const data = await res.json();
                resBox.innerHTML = '';
                data.roadmap.forEach((step, i) => {
                    resBox.innerHTML += `
                        <div class="bg-white p-8 rounded-[2.5rem] border border-slate-100 flex gap-8 items-start shadow-sm hover:border-indigo-600 transition-all text-left">
                            <div class="w-12 h-12 bg-indigo-950 text-white rounded-2xl flex items-center justify-center font-black shrink-0 shadow-xl">${i+1}</div>
                            <p class="text-slate-800 font-extrabold pt-2 text-lg leading-relaxed">${step}</p>
                        </div>
                    `;
                });
            } catch(e) { resBox.innerHTML = '<p class="text-red-500 font-black">Error: Check FastAPI logs.</p>'; }
        }

        // Input Listeners
        document.getElementById('chatField').addEventListener('keypress', (e) => { if(e.key === 'Enter') sendChatApi(); });
        function clearChatMessages() { document.getElementById('chatWindow').innerHTML = '<p class="text-center text-slate-300 text-[10px] font-bold py-10">New Session Cleared</p>'; }

from fastapi import FastAPI
from pydantic import BaseModel
from typing import List
from fastapi.middleware.cors import CORSMiddleware
import google.genai as genai

# -------- App --------

app = FastAPI()

# -------- CORS Middleware --------

app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# -------- Gemini Client --------
# Paste your Gemini API Key inside ""

client = genai.Client(api_key="AIzaSyD9kgwgIKc4GcrGSaP-AuzLvEKJLcZ_9YM")

# -------- Models --------

class ChatRequest(BaseModel):
    message: str


class RoadmapRequest(BaseModel):
    role: str


class AlumniCreate(BaseModel):
    name: str
    company: str
    role: str
    skills: str
    batch: int
    mentor_available: bool


# -------- Fake Database --------

alumni_db: List[AlumniCreate] = []

# Add one demo alumni

alumni_db.append(
    AlumniCreate(
        name="Nandini Singh",
        company="Google",
        role="AI Engineer",
        skills="Python, FastAPI, React",
        batch=2024,
        mentor_available=True,
    )
)

# -------- Home --------

@app.get("/")
def home():
    return {"message": "Alumni AI API running 🚀"}


@app.post("/chat")
def chat(request: ChatRequest):
    try:
        response = client.models.generate_content(
            model="gemini-2.5-flash",
            contents=f"You are an alumni mentor helping students. Question: {request.message}. Give simple advice."
        )

        return {
            "response": response.text
        }

    except Exception as e:
        return {"response": str(e)}
# -------- Career Roadmap --------

@app.post("/career-roadmap")
def career_roadmap(request: RoadmapRequest):
    try:
        response = client.models.generate_content(
            model="gemini-2.5-flash",
            contents=f"Provide a 5 step career roadmap for a {request.role}. Give only short steps."
        )

        steps = [s.strip() for s in response.text.split("\n") if s.strip()]

        return {"roadmap": steps[:5]}

    except Exception:
        return {"roadmap": ["Step 1: Learn basics", "Step 2: Build projects"]}


# -------- Add Alumni --------

@app.post("/alumni")
def add_alumni(alumni: AlumniCreate):
    alumni_db.append(alumni)
    return {"message": "Alumni added successfully"}


# -------- Get Alumni --------

@app.get("/alumni")
def get_alumni():
    return alumni_db


# -------- Mentor Match --------

@app.get("/mentor-match")
def mentor_match(skill: str):
    matches = [
        alum
        for alum in alumni_db
        if skill.lower() in alum.skills.lower() and alum.mentor_available
    ]

    return matches
